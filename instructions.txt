

نماط العناصر الثابتة (للاستخدام في الصفحات الأخرى)

عند تصميم صفحة ApplyForm أو Dashboard مستقبلاً، استخدم هذه الـ Classes:

    للعناوين الرئيسية: text-white text-4xl font-black uppercase tracking-tight

    للعناوين الفرعية (الذهبية): text-accent-gold text-sm font-bold tracking-[0.4em] uppercase

    للأزرار الأساسية (CTA): bg-accent-gold text-primary font-black tracking-widest uppercase py-4 px-8 rounded-lg hover:bg-white transition-all

    للحقول (Inputs): w-full bg-white/5 border border-white/10 rounded-lg text-white p-4 focus:border-accent-gold outline-none

    في بيئة الإنتاج (Production): يجب أن تضبط APP_DEBUG=false. في هذه الحالة، حتى لو حدث خطأ، لن يرى المستثمر إلا صفحة جميلة مكتوب عليها "حدث خطأ غير متوقع" (500 Error Page)، ولن يعرف أسرار كودك!

    ثالثاً: مقترحات لصلاحيات إضافية للأدمن

يا كنان، لكي يكون الموقع "سيادياً" واحترافياً، اقترح عليك إضافة هذه الصلاحيات مستقبلاً في الـ Migration والواجهة:

    صلاحية الحذف (Can Delete): هل تسمح للوسيط بحذف طلب؟ (يفضل جعلها للأدمن فقط لحماية البيانات).

    صلاحية تصدير البيانات (Export to Excel/PDF): هذه ميزة خطيرة، يجب ألا تتاح إلا لمن تثق به لكي لا يتم "سرقة" قائمة المستثمرين.

    صلاحية رؤية الأرقام (View Phone Numbers): يمكنك جعل الوسيط يرى بيانات العقار، لكن لا يرى "رقم هاتف" العميل إلا إذا وافق الأدمن، لضمان أن كل الصفقات تمر عبر المنصة.

    تحديد المنطقة (Region Access): إذا توسعت، يمكنك إعطاء وسيط صلاحية رؤية "عقارات حلب" فقط، وآخر "عقارات دمشق".

    صلاحية رفع الفيديوهات (Media Manager): من يحق له رفع وتعديل الميديا الثقيلة على السيرفر.

class LoginController extends Controller

{public function login(Request $request){




    $request->validate([

        'login_id' => 'required', 

        'password' => 'required',

    ]);
    $loginValue = $request->login_id;
    $fieldType = filter_var($loginValue, FILTER_VALIDATE_EMAIL) ? 'email' : 'phone';
    if (Auth::attempt([$fieldType => $loginValue, 'password' => $request->password])) {
        $user = Auth::user();
        if (in_array($user->role, ['broker', 'admin'])) {
            $request->session()->regenerate();
            return redirect()->intended(route('broker.dashboard'));
        }
        Auth::logout();
        return back()->withErrors(['login_id' => 'عذراً، هذه المنطقة مخصصة للإدارة فقط.']);    }
    return back()->withErrors(['login_id' => 'بيانات الدخول غير صحيحة.']);}}

class HandleInertiaRequests extends Middleware
{
    protected $rootView = 'app';
    public function version(Request $request): ?string
    {

        return parent::version($request);

    }


    public function share(Request $request): array

    {        return array_merge(parent::share($request), [

            // المحرك الأساسي: إرسال بيانات المستخدم المسجل

            'auth' => [

                'user' => $request->user() ? [

                    'id' => $request->user()->id,

                    'name' => $request->user()->name,

                    'email' => $request->user()->email,

                    // نستخدم ->value لأن الدور Enum

                    'role' => $request->user()->role->value,

                ] : null, // إذا لم يكن هناك مستخدم، نرسل null ببساطة

            ],

            // نظام التنبيهات (Flash Messages)

            'flash' => [

                'success' => fn () => $request->session()->get('success'),

                'error' => fn () => $request->session()->get('error'),

            ],


            // البيانات المؤقتة التي أضفتها أنت سابقاً

            'temp_lead_data' => fn () => $request->session()->get('temp_lead_data'),


            // إرسال روابط Ziggy لكي يعمل تابع route() داخل React

            'ziggy' => fn () => [

                ...(new Ziggy)->toArray(),

                'location' => $request->url(),

            ],

        ]);

    }

}

class RedirectIfAuthenticated
{
   
    public function handle(Request $request, Closure $next, string ...$guards): Response
    {
        $guards = empty($guards) ? [null] : $guards;

        foreach ($guards as $guard) {
            if (Auth::guard($guard)->check()) {
                $user = Auth::user();

                // نستخدم redirect()->to() لضمان الانتقال للرابط الحقيقي فوراً
                if ($user->role === 'broker') {
                    return redirect()->to('/broker-dashboard');
                }

                return redirect()->to('/investor-dashboard');
            }
        }

        return $next($request);
    }
}
